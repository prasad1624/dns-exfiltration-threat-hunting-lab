# pfSense Egress Filtering Rules

# Rule 1: Allow DNS ONLY to internal resolver 
Action: Pass
Protocol: UDP/TCP
Source: LAN (10.0.2.0/24)
Destination: 10.0.2.1
Port: 53
Description: Allow DNS to pfSense DNS resolver

# Rule 2: Block DNS to any other destination
Action: Block
Protocol: UDP/TCP
Source: LAN (10.0.2.0/24)
Destination: !10.0.2.1
Port: 53
Description: Block DNS to any server except internal resolver

# Rule 3: Block DNS on non-standard ports (tunneling evasion)
Action: Block
Protocol: UDP/TCP
Source: LAN (10.0.2.0/24)
Destination: any
Port: !53
Description: Block DNS on non-standard ports

# Rule 4: Default deny for all other traffic
